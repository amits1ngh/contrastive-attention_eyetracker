#Loading the Dataset
df_code<-read.csv2(file ='code_05.07.22.csv', dec = ",") #loading dataframe

#summing up the the path responses (successes)
df_code$sum_responses<-rowSums(df_code[, c('pre_obj','path1','path2','path3','mid_obj','path5','path6','post_obj')], na.rm = TRUE) 

#total (chances)
df_code <- df_code%>%mutate(n= case_when(path=='C1' ~ 5, path=='C2' ~ 5, path=='C3' ~ 8, path=='C4' ~ 8))

Factor adjustments and contrast coding

#converting the levels to factors
df_voice_cond_added_code$path_cond<- factor(df_voice_cond_added_code$path_cond, levels = c("Contrastive", "Non-Contrastive"))
df_voice_cond_added_code$voice_cond<- factor(df_voice_cond_added_code$voice_cond, levels = c("No Voice", "Assertion-Assertion", "Negation-Negation", "Negation-Assertion", "Assertion-Negation"))

Model fitting, logistic regression with vocie and path condition as fixed effects and random intercept for subjects. This was the best fit, a random slope for subject either for voice_cond and path_cond throws a singularity warning.

#logistic model fit
m.log_recall <- glmer(cbind(sum_responses, n-sum_responses) ~ voice_cond*path_cond + 
                 (1|SubjID), family = binomial,
               control=glmerControl(optimizer = "bobyqa"), data=df_voice_cond_added_code)
